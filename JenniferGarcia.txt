REATE TABLESPACE big_tbs_JennGarcia DATAFILE 
'JennGarcia1.dbf' SIZE 1G;

CREATE TABLESPACE QUIZ DATAFILE 
'QUIZ1.dbf' SIZE 200M,
'QUIZ2.dbf' SIZE 200M,
'QUIZ3.dbf' SIZE 200M
AUTOEXTEND ON NEXT 600K MAXSIZE 1000M
EXTENT MANAGEMENT LOCAL 
SEGMENT SPACE MANAGEMENT AUTO
ONLINE;

select * from SYS.DBA_TABLESPACES;
select * from DBA_DATA_FILES;

CREATE PROFILE PERFIL
LIMIT SESSIONS_PER_USER UNLIMITED
CPU_PER_SESSION UNLIMITED
IDLE_TIME 15
FAILED_LOGIN_ATTEMPTS UNLIMITED;

CREATE USER USUARIO
    IDENTIFIED BY PWD 
    DEFAULT TABLESPACE QUIZ 
    QUOTA UNLIMITED ON QUIZ 
    PROFILE PERFIL 
    PASSWORD EXPIRE;
    
GRANT CONNECT,RESOURCE TO USUARIO;
